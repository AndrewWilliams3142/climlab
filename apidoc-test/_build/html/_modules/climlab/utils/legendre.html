    Pnprime = {}
    Pnprime['0'] = 0
    Pnprime['1'] = P1prime(x)
    Pnprime['2'] = P2prime(x)
    Pnprime['3'] = P3prime(x)

    Pnprime['10'] = P10prime(x)
    Pnprime['12'] = P12prime(x)
    Pnprime['14'] = P14prime(x)
    return Pnprime
